    // 2. PROTOTYPE
    public class WorkerDrone : ICloneable
    {
        public string Model { get; set; }
        public int EnergyLevel { get; set; }

        public WorkerDrone(string model)
        {
            Model = model;
            EnergyLevel = 100;
        }

        public object Clone()
        {
            Console.WriteLine($"[Prototype] Клонування дрона моделi {Model}...");
            return this.MemberwiseClone();

    // 4. FACTORY
    public class Asteroid
    {
        public string ResourceType { get; set; }
    }

    public static class AsteroidFactory
    {
        public static Asteroid CreateAsteroid(string type)
        {
            Console.WriteLine($"[Factory] Створення астероїда типу: {type}");
            return type.ToLower() switch
            {
                "gold" => new Asteroid { ResourceType = "Золото" },
                "iron" => new Asteroid { ResourceType = "Залiзо" },
                _ => throw new ArgumentException("Невiдомий тип ресурсу")
            };
        }
    }

    // 6. FLYWEIGHT
    public class ParticleVisual
    {
        public string Color { get; }
        public string Texture { get; }
        public ParticleVisual(string color, string texture) { Color = color; Texture = texture; }
    }

    public class ParticleFactory
    {
        private Dictionary<string, ParticleVisual> _cache = new Dictionary<string, ParticleVisual>();
        public ParticleVisual GetVisual(string color, string texture)
        {
            string key = color + texture;
            if (!_cache.ContainsKey(key))
            {
                Console.WriteLine($"[Flyweight] Створення нової текстури: {key}");
                _cache[key] = new ParticleVisual(color, texture);
            }
            return _cache[key];
        }
    }

    // 7. PROXY
    public interface IOrbitalStrike { void Fire(); }

    public class RealOrbitalStrike : IOrbitalStrike
    {
        public void Fire() => Console.WriteLine("[Proxy] БУУУУМ! Орбiтальний удар успiшний! Ворога знищено.");
    }

    public class OrbitalStrikeProxy : IOrbitalStrike
    {
        private RealOrbitalStrike _realStrike;
        private int _playerEnergy;

        public OrbitalStrikeProxy(int energy) { _playerEnergy = energy; }

        public void Fire()
        {
            if (_playerEnergy >= 1000)
            {
                if (_realStrike == null) _realStrike = new RealOrbitalStrike();
                _realStrike.Fire();
                _playerEnergy -= 1000;
            }
            else
            {
                Console.WriteLine($"[Proxy] Помилка: Недостатньо енергiї для орбiтального удару (Потрiбно 1000, є {_playerEnergy}).");
            }
        }
    }

namespace BehavioralPatternsDemo

    // 1. CHAIN OF RESPONSIBILITY
    public abstract class Officer
    {
        protected Officer _nextOfficer;
        public void SetNext(Officer next) => _nextOfficer = next;
        public abstract void HandleSignal(int dangerLevel);
    }

    public class Lieutenant : Officer
    {
        public override void HandleSignal(int dangerLevel)
        {
            if (dangerLevel < 5) Console.WriteLine("[Chain] Лейтенант розiбрався з дрiбною загрозою");
            else if (_nextOfficer != null) _nextOfficer.HandleSignal(dangerLevel);
        }
    }

    public class Admiral : Officer
    {
        public override void HandleSignal(int dangerLevel) =>
            Console.WriteLine("[Chain] Адмiрал приймає командування! Загроза критична");
    }

    // 2. COMMAND
    public interface ICommand { void Execute(); }

    public class MoveCommand : ICommand
    {
        private string _unit;
        public MoveCommand(string unit) { _unit = unit; }
        public void Execute() => Console.WriteLine($"[Command] {_unit} перемiщується");
    }

    public class AttackCommand : ICommand
    {
        private string _unit;
        public AttackCommand(string unit) { _unit = unit; }
        public void Execute() => Console.WriteLine($"[Command] {_unit} атакує цiль");
    }

    public class CommandQueue
    {
        private Queue<ICommand> _queue = new Queue<ICommand>();
        public void AddCommand(ICommand cmd) => _queue.Enqueue(cmd);
        public void ExecuteAll() { while (_queue.Count > 0) _queue.Dequeue().Execute(); }
    }

        public void Update(string eventInfo) =>
            Console.WriteLine($"[Observer] Турель отримала координати: {eventInfo}. Вiдкриваю вогонь");

    public class RadarSystem
    {
        private List<IObserver> _turrets = new List<IObserver>();

        {
            Console.WriteLine("[Observer] Радар: Виявлено ворога у секторi 7");

    // 10. VISITOR
    public interface IVisitor { void Visit(Mech mech); void Visit(Base structure); }

    public class Mechanic : IVisitor
    {
        public void Visit(Mech mech) => Console.WriteLine("[Visitor] Механiк лагодить сервоприводи Меха");

    public interface IVisitable { void Accept(IVisitor visitor); }
    public class Mech : IVisitable { public void Accept(IVisitor visitor) => visitor.Visit(this); }
    public class Base : IVisitable { public void Accept(IVisitor visitor) => visitor.Visit(this); }
